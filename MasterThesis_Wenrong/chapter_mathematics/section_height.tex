\section{Height}

In this section, we will discuss the definition and properties of height of a formal group law.
Let $F(X,Y)$ be a commutative formal group law over ring $R$, fixed throughout this section.

From now on, we fix a prime number $p$ and assume that $R$ is of characteristic $p$,

\begin{definition}[Height of homomorphism]
    Let $F(X,Y), G(X,Y)$ be formal group laws over $R$, $f \in R[\![X]\!]$ be a homomorphism from
    $F(X,Y)$ to $G(X,Y)$, then the height of $f$, denoted by $\htt(f)$, is the largest number $h$
    such that
    \[f(X) = g(X ^ {p ^ h})\]
    for some power series $g(X)$ over $R$. In particular, if $f= 0$, then we say $\htt(f) = \infty$.
\end{definition}

\begin{remark}
    Let $m$ be a natural number prime to $p$, then $\htt([m]) = 0$, since
        \[[m](X) = m X + \cdots\]
\end{remark}

\begin{definition}[Height of formal group]
    Height of formal group law $F$ is defined to be the height of the $p$-series $[p]_F(X)$,
    denoted by $\htt (F)$.
    More precisely, let $v_n$ be the $p^n$-th coefficient of the $p$-series $[p](X)$. We say
    $F(X,Y)$ has height $n$, if $v_i = 0$ for all $i < n$ and $v_n \neq 0$ in $R$.
    In particular, we say $F(X,Y)$ has infinity height
    if $[p](X) = 0$.
\end{definition}

\begin{remark}
    In the Lurie's lecture note \cite{lurielecture12}, there is a slight modification
    in the definition of height of formal group law. We say $F(X,Y)$ has height $\ge n$,
    if $v_i = 0$ for all $i < n$
    And we say $F(X,Y)$ has height exactly $n$ if $v_n$ is invertible in $R$.
\end{remark}

\begin{remark}
    From the remark above, we have that $v_0 = p$. Then $F(X,Y)$ has height $\ge 1$ if and only if
    $p = 0$ in $R$, which is equivalent to either $R$ of characteristic $p$ or $R$ is trivial. And
    $F(X,Y)$ has height exactly $0$ if and only if $p$ is invertible in $R$.
\end{remark}

\begin{example}
    The $p$-series for multiplicative formal group
    $\bfG_m(X,Y) = (1 + X)(1 + Y) - 1$ is $[p](X) = (1 + X)^p - 1 = X ^ p$, then $\bfG_m(X,Y)$
    has height exactly $1$. For the additive formal group law $\bfG_a(X,Y)$ the $p$-series
    $[p](X) = p X = 0$, then $\bfG_a(X,Y)$ has infinity height.
\end{example}

\begin{proposition}
    Let $F(X,Y)$ and $G(X,Y)$ be formal group laws over ring $R$, then $F(X,Y)$ and $G(X,Y)$ has
    the same heights.
\end{proposition}

\begin{proof}
    Let $g(X,Y)$ be a formal group isomorphism from $F(X,Y)$ to $G(X,Y)$,
    $[p]_F(X)$ and $[p]_G(X)$ be $p$-series for $F(X,Y)$ and $G(X,Y)$ respectively.
    Note that
    \begin{equation*}
        [p]_G (X) = (g \circ [p]_F \circ g^{-1}) (X)
    \end{equation*}
    It follows that the heights of $F(X,Y)$ and $G(X,Y)$ is the same.
\end{proof}

\begin{remark}
    The additive and multiplicative formal group are isomorphic over a ring $R$ of
    characteristic $p$.
\end{remark}

\begin{proposition}
    Let $F,G,H$ be formal group laws over $R$, and let
    \[F \stackrel{f}{\longrightarrow} G \stackrel{g}{\longrightarrow} H \]
    be a chain of homomorphism defined over $R$. Then
    \[\htt(g \circ f) = \htt(f) + \htt (g) \]
\end{proposition}

\begin{proof}
    Denote that $f (X) = f_1 (X ^ {p ^ {\htt (f)}})$ and $g(X) = g_1 (X ^ {p ^ {\htt (g)}})$. Then
    \[(g\circ f) (X) = g_1 (f_1 (X ^ {p ^ {\htt (f)}}) ^ {\htt (g)})
        = g_1 (\tilde{f}_1(X ^ {\htt(f) + \htt(g)}))\]
    where $\tilde{f}_1 (X) = \frob ^ {\htt(g)}_* f_1(X)$, namely raising each coefficient of
    $f_1$ to $p^{\htt(g)}$-power.
\end{proof}

\begin{proposition}
    Let $F, G$ be formal group law over $R$, $f$ be a homomorphism from $F$ to $G$ defined over
    $R$, then
    \begin{enumerate}
        \item If $f'(0) = 0$, then $f(X) = f_1(X ^ p)$ for some $f_1(X) \in R[\![X]\!]$. That is
            $\htt(f) \ge 1$.
        \item Write $f(X) = g(X^ {p^{\htt(f)}})$. Then $g'(0) \neq 0$.
    \end{enumerate}
\end{proposition}

\begin{proof}
    The proof of this proposition need to use results that we discuss in the section of
    invariant differential.


    For the first part of this proposition, Let $\omega_F, \omega_G$ be the normalized invariant differential of $F$ and $G$
    respectively. According to the corollary \ref{cor: omega_hom_a}, we have that
    \begin{align*}
        0 &= f'(0) \omega_F \\
         & = \omega_G \circ f
    \end{align*}
    And if we denote that $\omega_G(X) = P_G(X) dX$, then we have that
    \[ \omega_G \circ f(X) = \omega_G(f(X)) = P_G(f(X)) f'(X) dX. \]
    Then we have that $f'(X) = 0$ since $P_G(X)$ is of form $(1 + \cdots)$.

    For the second part of this proposition, we first denote $q = p ^ {\htt(f)}$
    and $F_q = \phi_* F$, where $\phi$ is a ring endomorphism,
    mapping $a \mapsto a ^ {q}$. Then $F_q$ is also a formal group law. Now we claim that
    $g(X)$ is a homomorphism from $F_q$ to $G$. We prove this claim by following calculation:
    \begin{align*}
        g(F_q(X,Y)) & = g (F(X,Y) ^ q)\\
        & = g (X ^ q) \circ F \\
        & = f(X) \circ F \\
        & = g \circ F
    \end{align*}
    The first equation is guaranteed by the characteristic of $R$.
    If $g'(0) \neq 0$, then due to the first assertion of the proposition, there is a
    $g_1(X)$ such that $g(X) = g'(X ^ p)$. It follows that $f(X) = g_1(X^{p^{\htt(f) + 1}})$, which
    contradicts to the definition of height of $f$.
\end{proof}

\begin{theorem}[Lazard]
    Let $k$ be an algebraically closed field with characteristic $p$, then two formal group laws
    $F(X,Y), G(X,Y)$ over $k$ is isomorphic if and only if they have the same heights.
\end{theorem}

\begin{proof}
    Proof can be found in the Lurie's lecture note \cite{lurielecture13} and \cite{lurielecture14}.
\end{proof}
