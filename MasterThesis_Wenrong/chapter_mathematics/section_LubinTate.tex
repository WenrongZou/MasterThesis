\section{Definition and Properties}

Let $K$ be a field complete with respect to a discrete valuation, with finite
residue field. For instance, $K$ is a nonarchimedean local field. Let $\calO_K$ be
the ring of integers of $K$, let $q$ be the cardinality of the residue
field, and let $\pi$ be a uniformizer of $\calO_K$.

\begin{definition}[{\cite[Chap.~6 Definition 3.3]{cassels1967algebraic}}]
    Let $\calF_\pi$ denote the set of power series $f(X) \in \calO_K[\![X]\!]$ which
    satisfy the two conditions:
    \begin{align}
        f(X) &\equiv \pi X \ (\modd \ \deg 2) \\
        f(X) &\equiv X ^ q \ (\modd \ \pi)
        \label{Lubin_Tate_condition}
    \end{align}

\end{definition}

\begin{lemma}[{\cite[Chap.~6 Proposition 5]{cassels1967algebraic}}]
    Let $f(X)$ and $g(X)$ be elements of $\calF_\pi$ and let $L(X_1, \ldots, X_n)
    = \sum_{i = 1}^n a_i X_i$ be a linear form with coefficients in $\calO_K$.

    Then there
    exists $F(X_1, \ldots, X_n)$ with coefficients in $\calO_K$ such that
    \begin{align}
        F (X_1, \ldots, X_n) &\equiv  L(X_1, \ldots, X_n) (\modd \ \deg 2) \\
        f(F(X_1, \ldots, X_n)) &= F(g(X_1), \ldots, g(X_n))
    \end{align}
    \label{lemma: constructive_lemma_lubin_Tate}
\end{lemma}

\begin{proof}
    We use induction to prove that for all $k \in \NN$, there exists a unique multivariate
    polynomial $F_k(X_1, \ldots, X_n)$ of degree $k$ and
    \begin{align*}
        F_k(X_1, \ldots , X_n) &\equiv L (X_1, \ldots, X_n) (\modd \deg 2) \\
        f(F_k(X_1, \ldots , X_n)) &\equiv F_k(g(X_1), \ldots, g(X_n)) (\modd \deg (k + 1))
    \end{align*}\par
    The existence and uniqueness of $F(X_1, \ldots, X_n)$ then follow from this. \par
    For $k = 1$, we take $F_1(X_1, \ldots, X_n) = L(X_1, \ldots, X_n)$. \par
    Now assume that the result holds for $k$. Write $f (F_k(X_1,\ldots, X_n )) \equiv
        F_k(g(X_1), \ldots, g(X_n)) + E_{k+1} (\modd \deg (k+2))$. According to the induction
        hypothesis, $E_{k+1}$ is homogeneous of degree $k + 1$. \par
    On the other hand, since the uniqueness of $F_k(X_1, \ldots, X_n)$, then
    $F_{k+1}(X_1,\ldots,X_n)$ should be of the form $F_{k+1}(X_1,\ldots, X_n) =
        F_k(X_1, \ldots, X_n) + \phi_{k+1}(X_1, \ldots, X_n)$, where
        $\phi_{k+1}(X_1, \ldots, X_n)$ is homogeneous of degree $k+1$, due to the definition of
        $\calF_\pi$. Then we have
    \begin{align*}
        f(F_{k+1}(X_1, \ldots, X_n)) &\equiv F_k(g(X_1), \ldots, g(X_n)) +
        \pi \phi_{k+1}(X_1, \ldots, X_n) (\modd \deg (k+2)) \\
        F_{k+1}(g(X_1), \ldots, g(X_n)) &\equiv  F_k(g(X_1), \ldots, g(X_n)) +
        \pi ^ {k+1} \phi_{k+1} (X_1, \ldots, X_n) (\modd \deg (k + 2))
    \end{align*}
    The last congruence follows from the fact that $\phi_{k+1}$ is homogeneous of degree $k+1$, i.e.
    $\phi_{k+1} (g(X_1), \ldots, g(X_n)) = \pi ^ {k+1} \phi_{k+1} (X_1, \ldots, X_n)$.
    Therefore, we conclude that
    \begin{equation}
        E_{k+1} + (\pi- \pi^{k+1}) \phi_{k+1} (X_1, \ldots, X_n) = 0
    \end{equation}
    Then $\phi_{k+1}(X_1, \ldots, X_n) = - E_{k+1}/ (\pi - \pi^{k+1})$.
\end{proof}

\begin{definition}
    For any $f \in \calF_\pi$, we let $F_f(X,Y)$ be the unique solution of
    \begin{align}
        F_f(X,Y) &\equiv X + Y (\modd \ \deg 2) \\
        f(F_f(X,Y)) &= F_f(f(X), f(Y))
    \end{align}
\end{definition}


\begin{theorem}[Lubin-Tate Formal Group Laws {\cite[Chap.~6 Proposition 1]{cassels1967algebraic}}]
    For power series $f \in \calF_\pi$, we have
    \begin{align}
        F_f(X,Y) &= F_f(Y,X) \label{eq: lubin_Tate_comm} \\
        F_f(F_f(X,Y),Z) &= F_f(X, F_f(Y,Z)) \label{eq: lubin-Tate_assoc}
    \end{align}
     As a result, $F_f(X,Y)$ is a commutative formal group law over $\calO_K$.

    \label{thm: Lubin_Tate}
\end{theorem}

\begin{proof}
    For equation \ref{eq: lubin_Tate_comm}, let $G (X,Y)= F(Y,X)$, then $G(X,Y)
    \equiv X+Y (\modd \deg 2)$ and
    \begin{equation*}
        f(G(X,Y)) = f(F(Y,X)) = F(f(Y),f(X)) = G(f(X),f(Y))
    \end{equation*}
    Therefore, by the uniqueness in Lemma \ref{lemma: constructive_lemma_lubin_Tate},
    $F(X,Y) = G(X,Y) = F(Y,X)$. \par
    For equation \ref{eq: lubin-Tate_assoc}, let $G_1(X,Y,Z) = F_f(F_f(X,Y),Z)$ and
    $G_2(X,Y,Z) = F_f(X,F_f(Y,Z))$, and we can check that $G_i(X,Y,Z) \equiv X+Y+Z (\modd \deg 2)$ and
    \begin{equation*}
        f(G_i(X,Y,Z)) = G_i(f(X),f(Y),f(Z))
    \end{equation*}
    Then by the uniqueness in Lemma \ref{lemma: constructive_lemma_lubin_Tate},
    we conclude the desired equation.
\end{proof}


\begin{definition}
    For each $a \in \calO_K$ and $f, g \in \calF_\pi$, let $[a]_{f,g} (X)$ be the unique
    solution of
    \begin{align}
        [a]_{f,g} (X) &\equiv aX (\modd \deg 2) \\
        f([a]_{f,g}(X)) &= [a]_{f, g}(g(X))
    \end{align}
    In particular, denote $[a]_{f}$ instead of $[a]_{f,f}$ to simplify the notation.
\end{definition}

\begin{theorem}[{\cite[Chap.~6 Proposition 2]{cassels1967algebraic}}]
    For each $a \in \calO_K$ and $f, g \in \calF_\pi$, the power series $[a]_{f,g}(X)$ is a
    formal group homomorphism from $F_g(X,Y)$ to $F_f(X,Y)$, i.e.
    $[a]_{f,g} F_g(X,Y) = F_f([a]_{f,g}(X), [a]_{f,g} (Y))$.
\end{theorem}

\begin{proof}
    For simplicity, we denote $[a]_{f,g}$ as $h$. We have $h(F_g(X,Y)) \equiv aX+aY
    (\modd \deg 2)$, $F_f(h(X), h(Y)) \equiv aX+ aY (\modd \deg 2)$ and
    \begin{align*}
        h\circ F_g(g(X),g(Y)) &= h\circ g(F_g(X,Y)) = f \circ h (F_f(X,Y)) \\
        F_f(h(g(X), h(g(Y)))) &= F_f(f(h(X)), f(h(Y))) = f \circ F_f(h(X), h(Y))
    \end{align*}
    Then the result follows from uniqueness in Lemma \ref{lemma: constructive_lemma_lubin_Tate}.
    \end{proof}

\begin{theorem}[{\cite[Theorem 1]{LubinTate1965}}]
    Let $a, b \in \calO_K$, $f, g, h \in \calF_\pi$, then $[a]_{f,g} ([b]_{g,h}(X)) = [ab]_{f,h} (X) $.
\end{theorem}

\begin{proof}
    We have $[a]_{f,g} ([b]_{g,h}(X)) \equiv abX (\modd \deg 2)$ and
    \begin{equation*}
        f \circ ([a]_{f, g} \circ [b]_{g,h}) = [a]_{f,g} \circ (g \circ [b]_{g,h}) = [a]_{f,g}\circ [b]_{g,h} \circ h
    \end{equation*}
     Then  we have $[ab]_{f,h}$ and $[a]_{f,g} \circ [b]_{g,h}$ are solutions satisfying these conditions. By the uniqueness of Lemma \ref{lemma: constructive_lemma_lubin_Tate}, they coincide.
\end{proof}

\begin{theorem}[{\cite[Theorem 1]{LubinTate1965}}]
    Let $a, b \in \calO_K$, $f, g \in \calF_\pi$, then
    $F_f([a]_{f,g}(X), [b]_{f,g}(X)) = [a+b]_{f,g} (X) $.
\end{theorem}

\begin{proof}
    Notice that $F_f([a]_{f,g}(X), [b]_{f,g}(X)) \equiv (a+b)X (\modd \deg 2)$ and
    \begin{equation*}
        f(F_f([a]_{f,g}(X), [b]_{f,g}(X))) = F_f(f\circ [a]_{f,g}(X), f\circ [b]_{f,g}(X))
            = F_f([a]_{f,g}\circ f(X), [b]_{f,g}\circ f(X))
    \end{equation*}
    Therefore, by the uniqueness in Lemma \ref{lemma: constructive_lemma_lubin_Tate}, we have
        $F_f([a]_{f,g}(X), [b]_{f,g}(X)) = [a+b]_{f,g} (X) $.
\end{proof}

\begin{theorem}[{\cite[Chap.~6 Proposition 4]{cassels1967algebraic}}]
    Let $f,g\in \calF_\pi$, then $F_f$ is isomorphic to $F_g$, and there is a unique strong
    isomorphism from $F_f$ to $F_g$. This means that the Lubin-Tate formal group $F_f$ is
    independent of $f \in F_\pi$, and depends only on $\pi$.
    \label{thm: Lubin_Tate_independet_f_g}
\end{theorem}

\begin{proof}
    Take a unit $a \in \calO_K$, then $[a]_{f,g} \circ [a^{-1}]_{g,f} = [1]_f$ and
    $[a ^ {-1}]_{g,f} \circ [a]_{f,g} = [1]_g $. And the unique strong isomorphism is
    $[1]_{f,g}$.
\end{proof}

\begin{remark}
    Let $L$ be an algebraic extension of $K$, let $\gothm_L$ be the maximal ideal in the
    ring of integers of $L$. Given elements $x_1, \ldots, x_n \in \gothm_L$ and a formal
    series $G(X_1, \ldots, X_n) \in \calO_K[\![X_1, \ldots, X_n]\!]$, the series
    $G(x_1, \ldots, x_n)$ converges to an element in $\gothm_L$ if the constant term of
    $G(X_1, \ldots, X_n)$ is zero. By above theorem, for each power series $f\in \calF_\pi$ this
    determines an $\calO_K$-module structure on the set $\gothm_L$. To be precisely, addition
    and scalar multiplication are defined by
    \begin{align}
        x + y &= F_f(x,y) \\
        ax &= [a]_f (x)
    \end{align}\par
    \label{rem: A-module}
\end{remark}

\section{Applications}
Fix an algebraic closure $\overline{K}$ of $K$, and let $K^{sep}$ be a separable closure. Given
a $f\in \calF_\pi$, for each integer $n \ge 1$, let $f^n(X)$ be $n$-th iterate of $f(X)$.
Let $\Lambda_{f,n}$ be the set of all roots of $f^n(X)$ in $\overline{K}$ and let
$K(\Lambda_{f,n})$ be the field generated over $K$ by these elements.
Moreover, we have the fact that for any $f,g\in \calF_\pi$, $\lambda \in \Lambda_{f,m}$ if and only
if $[1]_{g,f} (\lambda) \in \Lambda_{g,m}$. This means that the field $K(\Lambda_{f,n})$ is independent of $f$ and
depends only on uniformizer $\pi$. Therefore, we denote $K(\Lambda_{f,n})$ as $L_{\pi,n}$.
The same argument holds for $K(\Lambda_f)$.
Additionally, let $\Lambda_f = \bigcup_{n=1}^\infty \Lambda _{f,n} $, $L_\pi = K(\Lambda_f)$,
$G_{\pi,n} = \Gal(L_{\pi,n}/K)$, $G_\pi = \varprojlim G_{f,n}$. \par
According to the remark $\ref{rem: A-module}$, $F_f$ defines a $\calO_K$ structure on
$\gothm_{K^{sep}}$ and we denote this $\calO_K$-module as $M_f(\Ksep)$. Then $\Lambda_{f,n}$ is those
elements in $\lambda \in \gothm_{K^{sep}}$ such that $\pi ^n \lambda = 0$.

\begin{theorem}[{\cite[Chap.~6 Theorem 3]{cassels1967algebraic}}]
    The $\calO_K$-module $\Lambda_{f,n}$ is isomorphic to $\calO_K/ \pi^n \calO_K$ and
    $\calO_K$-module $\Lambda_f$ is isomorphic to $K / \calO_K $.
    \label{thm: Lam_iso_K_quot_O}
\end{theorem}

In order to prove this theorem, we need a result about divisible modules over a PID.

\begin{theorem}[Structure theorem of divisible module over PID {\cite{matlis1958injective}}]
    Let $R$ be a principal ideal domain and $K$ be its field of fractions. Then every divisible
    $R$-module is isomorphic to a direct sum of $K$ and copies of the injective hulls
    of the simple module $R/ (p)$, varying over prime elements $p\in R$. \par
    Let $M$ be a divisible $R$-module, then $M$ can be written as follows:
    \[M \cong (\bigoplus_{i \in I} K) \oplus (\bigoplus_{p \ prime} \bigoplus_{j \in J_p}
        E(R / (p)))  \]
    \begin{itemize}
        \item $I$ is an index set measuring how many copies of torison free part in $M$ are needed.
        \item $p$ varies over all prime elements in $R$.
        \item $J_p$ is an index set measuring how many copies of $p$-primary torison part
                $E(R/(p))$ are needed for each $p$.
        \item $E(R/(p))$ is the injective hull of $R/(p)$. Over PID, this is isomorphic to
                $K/R_{(p)}$.
    \end{itemize}
\end{theorem}

\begin{proof}
   The proof of the structure theorem for divisible modules can be found in Chapter 3 of
   $\cite{lam1999lectures}$ or Theorem 2.5 and Proposition 3.1 in \cite{matlis1958injective}.
\end{proof}

\begin{proof}[Proof of theorem $\ref{thm: Lam_iso_K_quot_O}$]
    The first claim follows by the fact that $\Lambda_{f,n}$ is the $\pi^n$-torsion of
        $\calO_K$-module $M_f(\Ksep)$. \par
    According to theorem \ref{thm: Lubin_Tate_independet_f_g}, we are free to choose
    $f\in \calF_\pi$ as we please. We take $f(X)= \pi X + X ^ q$. Since
    $f'(X) = q X^{q-1} + \pi \neq 0$, then $\Lambda_{f, 1}$ consists of $q$ distinct roots
    of the $X^q + \pi X = 0 $. For any given $\alpha \in \gothm_{K^{sep}}$, since
    $\pi X + X ^ q = \alpha$ is separable, then $\Lambda_{f,1} \subset \gothm_{\Ksep}$. This means
    that for any given $\alpha \in \gothm_\Ksep$, there exists a $\beta$ such that
    $\pi \cdot \beta = \alpha$ in $M_f(\Ksep)$. It follows that for any given $r \in \calO_K$,
    $r M_f(\Ksep) = M_f (\Ksep)$ which means that $M_f(\Ksep)$ is divisible $\calO_K$-module. Then
    $\Lambda_f$ is a divisible torsion $\calO_K$-module. Since $\calO_K$ is a PID, then we know that
    $\Lambda_f$ is isomorphic to $K/ \calO_K$. Indeed, $\Lambda_f$ is torsion module
    then there is no copies of $K$. Since $\pi$ is the only prime element in $\calO_K$ and
    $(\calO_K) _ {(\pi)} \cong \calO_K$, then $\Lambda_f$ is a copy of $K/ \calO_K$. Moreover,
    $\Lambda_{f,1}$ is a one dimensional vector space over residue field $k$ (since $\Lambda_{f,1}$
    has the same cardinality of residue field $k$), then $\Lambda_f \cong K / \calO_K$. Last claim
    follows by the fact that $|J| = \dim_k M_f(\Ksep)[\pi] = \dim_k \Lambda_{f,1} = 1$, where $J$
    is the index set of copies of $K/ \calO_K$.
\end{proof}

There is a natural homomorphism $\Gal (\Ksep/ K) \to \End(\Lambda_f)$, this follows from
the action of elements in the Galois group on roots of a polynomial. Moreover,
since $\Lambda _f \cong K / \calO_K$ and $\End(K/\calO_K) \cong \calO_K$, then it induces a
homomorphism $\Gal(\Ksep/ K) \to \Aut(K / \calO_K) \cong U_K$.

\begin{theorem}[{\cite[Chap.~6 Theorem 3]{cassels1967algebraic}}]
    The natural homomorphism $\Gal(\Ksep/K) \to \Aut (\Lambda_f)$ is an isomorphism.
\end{theorem}

\begin{proof}
    This map is injective by definition; it remains to prove that it is surjective.
    We have an injection $\Gal(L_{\pi,n}/ K) \to U_K/ U _K^n$, where $U_K = 1 + \pi ^ n \calO_K$.
    Let $\alpha \in \Lambda_{f,n}$ be a primitive element, namely those elements such that
    $[\pi ^ {n - 1}]_f \alpha \neq 0$ or $\pi ^ {n - 1} \cdot \alpha \neq 0$ in $M_f(\Ksep)$.
    To be precise, $\alpha$ is one of the roots of polynomial $f^{(n)}(X)/ f^{(n-1)}(X)$,
    where $f^{(n)}(X)$ is $n$-th iterate of $f(X)$.
    Since $f(X)/X = X ^ {q -1} + \pi$, then
    \[ \frac{f^{(n)}(X)}{f ^ {(n -1)}(X)} = f^ {(n -1)}(X)^{q-1} + \pi \]
    By the Eisenstein criterion, this is an irreducible polynomial with degree $q ^ n - q ^{n -1}$.
    Thus, according to the Galois theory, the order of $\Gal(L_{\pi,n}/ K)$ is at least
    $(q-1) q ^ {n-1}$. On the other hand, $U_K/ U_K ^ m$ has exactly $(q-1)q ^ {n -1}$
    elements. Therefore, $\Gal(L_{\pi,n}/K) \cong U_K/ U_K^n$. Eventually, we have that
    \[ \Gal(\Ksep/K) \cong \varprojlim \Gal(L_{\pi,n}/ K) \cong \varprojlim U_K / U_K^n \cong U_K. \]
\end{proof}

\begin{corollary}[{\cite[Chap.~6 Theorem 3]{cassels1967algebraic}}]
    The element $\pi$ is a norm from $L_{\pi,n}$.
\end{corollary}

\begin{proof}
    Let $\alpha \in \Lambda_{f,n}$ be a primitive element, then $f^{(n)}(X)/ f^{(n-1)}(X)$ is a
    monic polynomial with constant coefficient $\pi$, then the norm $-\alpha$ is $\pi$.
\end{proof}

Let $\Kun$ be the maximal unramified extension of $K$. Since $L_\pi$ is a totally ramified extension
of $K$, $L_\pi$ and $\Kun$ are linearly disjoint. As a result, the Galois group
$\Gal(L_\pi \Kun/K)$ is a product of $\Gal(L_\pi/K)$ and $\Gal(\Kun/K)$. Given a uniformizer
$\pi \in \calO_K$, we define a group homomorphism $r_\pi : K^\times \to \Gal(L_\pi \Kun / K)$ such that
\begin{itemize}
    \item $r_\pi (\pi)$ acts as identity on $L_\pi$ and acts as Frobenius automorphism $\sigma$ on
        $\Kun$.
    \item Given a $u \in U_K$, we define $r_\pi (u)$ to act as $[u^{-1}]_f$ on $L_\pi$ and
        acts as identity on $\Kun$.
\end{itemize}
Next, one can prove that the field $L_\pi$ and the homomorphism $r_\pi$ are independent of the
uniformizer $\pi$, and that $L_\pi$ is a maximal abelian extension of $K$. Moreover, $r_\pi$ coincides with
the local reciprocity map in local class field theory, i.e. for all $a \in \calO_K$,
$r_\pi(a) = (a, L_\pi \Kun / K ^ \times)$. The references of these can be found in
\cite{LubinTate1965} and \cite{cassels1967algebraic}.

\section{Connection with functional equation lemma}

In this section, we discuss the connection between the formal group laws obtained from the
functional equation lemma and Lubin--Tate formal group laws. Throughout this section
we fix a nonarchimedean local field $K$, let $\calO_K$ be its ring of integers,
$\pi$ be a uniformizer, $q$ be the cardinality of the
residue field of $\calO_K$.

\begin{proposition}
    The Lubin-Tate formal group laws $F_f$ associated to $f\in \calF_\pi$ are in one-to-one
    correspondence with formal group laws obtained by the functional equation with
    parameters $I = (\pi), \sigma = \id, s_1 = \pi^{-1}$ and $\forall i \ge 2, s_i = 0$.
    The functional equation in the sense of these parameters is
    \[ f_g(X) = g(X) + \pi^ {-1}f_g(X ^ q). \]
    And this correspondence is given by
    \[g(X) \mapsto f_g^{-1}(\pi f_g(X)) \in \calF_\pi. \]
\end{proposition}

\begin{proof}
    The integrality of the coefficients in $f_g^{-1}(\pi f_g(X))$ follows from the
    functional equation lemma. Multiplying both sides of the functional equation by $\pi$,
    we have that
    \[ \pi f_g(X) = \pi g(X) + \pi^{-1} \pi f_g(X). \]
    Then $\pi f_g(X) = f_{\pi g}(X)$. According to the second part of the functional
    equation lemma, we have that
    \[f_g^{-1} (\pi f_g(X)) = f_g^{-1} (f_{\pi g}(X)) \in \calO_K[\![X]\!]. \]
    The formal group law obtained by the functional equation lemma is
    \[ F(X,Y) = f_g^{-1} (f_g(X) + f_g(Y)) . \]
    By the existence and uniqueness in the theorem \ref{thm: Lubin_Tate}, it is enough
    to check that $f_g^{-1}(\pi f_g(X))$ is a homomorphism for $F(X,Y)$. This can be done as follows:
    \begin{align*}
        f_g^{-1} (\pi f_g(F(X,Y))) &= f_g^{-1} (\pi f_g(f_g^{-1} (f_g(X) + f_g(Y)))) \\
            & = f_g^{-1} (\pi f_g(X) + \pi f_g(Y)) \\
            & = F(f_g^{-1}(\pi f_g(X)), f_g^{-1}(\pi f_g(Y))).
    \end{align*}
\end{proof}
