\section{Commutativity theorem}
In this section, we are going to prove that `almost all' one dimensional formal group
is commutative. To be more precisely,

\begin{theorem}
    Let $R$ be a ring, then every one dimensional formal group law over ring $R$ is commutative if
    and only if $R$ has no element $a \neq 0$ that are nilpotent and
    $\ZZ$-torsion simultaneously.
    \label{comm_thm}
\end{theorem}

Before we start to prove the theorem, we should discuss some lemmas first.

\begin{lemma}
    Let $F(X,Y)$ be a non-commutative one dimensional formal group law over a ring
    $R$, then there is a nonzero homomorphism of $F(X,Y)$ into $\bfG_m(X,Y)$ or into
    $\bfG_m(X,Y)$.
    \label{lem: non_com_hom}
\end{lemma}

\begin{proof}
    For simplicity, we denote $F(X,Y)$ as $X +_F Y$ and the inverse $i(X)$ as $-_F X$.
    Let $H(X,Y) = X + _F Y + (-_F X)$, we have $H(0,Y) = Y$. And since $F(X,Y)$ is non-commutative,
    then
    \begin{equation}
        H(X,Y) = Y + \sum _{i=1}^\infty r_n(X) Y ^ n
    \end{equation}
    with $r_n(X) \neq 0$ for some $n$. Since $H(0,Y) = Y$, we have $r_n(0) = 0$ for all $n$. Due
    to the associativity of $F(X,Y)$, we have
    \begin{equation}
        H(X_1,H(X_2,Y)) = H(X_1+_F X_2, Y)
        \label{eq: H eq}
    \end{equation}
    Let $i$ be the least integer such that $r_i(X) \neq 0$. \\
    For the case $i = 1$, then
    \begin{equation*}
        H(X,Y) \equiv (1+ r_1(X)) Y\quad  (\modd Y ^ 2)
    \end{equation*}
    And combine with the equation $\ref{eq: H eq}$, we have

    \begin{equation}
        H(X_1, H(X_2,Y)) \equiv (1 + r_1(X_1)) H(X_2,Y) \equiv (1+ r_1(X_1)) (1 + r_1(X_2)) Y
        \quad (\modd  Y ^ 2)
    \end{equation}
    \begin{equation}
        H(X_1+_F X_2, Y) \equiv (1+ r_1(X_1+_F X_2)) Y \quad (\modd Y ^ 2)
    \end{equation}
    Therefore, $(1 + r_1(X_1)) (1+ r_1(X_2)) = 1 + r_1(X_1 + _F X_2)$, i.e.
    \begin{equation}
        r_1(X_1+_F X_2) = r_1(X_1) + r_1(X_2) + r_1(X_1) r_1(X_2).
    \end{equation}
    This means $r_1(X)$ is a formal group homomorphism from $F$ to $\bfG_m$. \par
    For the case $i > 1 $, then we have
    \begin{equation*}
        H(X,Y) \equiv Y + r_i(X) Y ^ i \quad (\modd Y ^ {i + 1})
    \end{equation*}
    According to the equation $\ref{eq: H eq}$, we have
    \begin{equation}
        H (X_1, H(X_2, Y)) \equiv H(X_2, Y) + r_i(X_1) H(X_2, Y) ^ i \equiv Y + r_i(X_2)Y ^ i +
            r_i(X_1) Y ^ i \quad (\modd Y ^ i).
    \end{equation}
    \begin{equation}
        H (X_1+_F X_2, Y) \equiv Y + r_i(X_1+_F X_2) Y ^ i \quad (\modd Y ^ {i + 1}).
    \end{equation}
    Therefore, $r_i(X_1+_F X_2) = r_i(X_1) + r_i(X_2)$, which means $r_i(X)$ is a formal group
    homomorphism from $F$ to $\bfG_a$.
\end{proof}

\begin{lemma}
    Let $R$ be a integral domain and let $F(X,Y), F'(X,Y)$ be one dimensional formal group laws
    over ring $R$. Suppose there exist a homomorphism
    $\alpha(X): F(X,Y) \to F'(X,Y)$ and $F'(X,Y)$ is commutative, then $F(X,Y)$ is
    commutative.
    \label{lem: target_com}
\end{lemma}

\begin{proof}
    Let $r$ be the order of $\alpha (X) = \sum_{n=1} ^ \infty a_n X^n$, i.e. the least $n$ such
    that $a_n$ does not equal to zero. Let $C(X,Y) = X +_F Y +_F (-_F X) +_F (-_F Y) $ be the
    commutator of $X$ and $Y$ with respect to $F$. \par
    We can prove that a formal group law $F$ is commutative if and only if the commutator
    $C(X,Y)$ with respect to $F$ is zero. \par
    Since $F'$ is commutative, then \[ \alpha (C(X,Y)) = 0.\]
    Suppose $F$ is non-commutative, i.e. $C(X,Y) \neq 0$. Let $C_n(X,Y)$ is the homogeneous
    component in $C(X,Y)$ with degree $n$, and $m$ be the least integer such that
    $C_m(X,Y)$ is non-zero. We have
    \[ \alpha (C(X,Y)) \equiv a_r C_m(X,Y) ^ r \quad (\modd \deg (mr + 1)) \]
    Since $a_r \in R$ and $R$ is a integral domain, then this term is non-zero, which contradicts
    to the assumption of $\alpha (C(X,Y))= 0$.
\end{proof}

\begin{proof}[Proof of Commutative theorem \ref{comm_thm}]
    First, we prove the necessity of the condition on $R$.\par
    Let $0 \neq a \in R$ be an element in $R$ is a nilpotent and $\ZZ$-torsion, then there is
    formal group law $F(X,Y)$ over $R$ such that $F(X,Y)$ is non-commutative. \par
    Let $n$ be the least integer such that $n a = 0$, then $n>1$ and there exists a prime number
    $p$, such that $p \mid n$. Let $b = \frac{n}{p} a$, then $b\neq 0$ and $p b = 0$, since $n$ is
    minimal. Let $m$ be the least integer such that $b^m= 0$ and $c = b^{m-1}$. Therefore, this $c$
    satisfies $c \neq 0, c ^ 2 = 0, p c = 0$. Denote $F(X,Y) = X + Y + c X Y ^ p$, then
    \begin{equation*}
        F(F(X,Y), Z) = X + Y + Z + c(X Y^p+ XZ^p + YZ^p) = F(X,F(Y,Z))
    \end{equation*}
    Therefore, $F(X,Y)$ is a non-commutative formal group law over $R$. \par
    Secondly, we prove sufficiency for the case $R$ is an integral domain. Let $F(X,Y)$ be a formal
    group law over $R$, where $R$ is an integral domain. Suppose that $F(X,Y)$ is non-commutative,
    then by
    lemma $\ref{lem: non_com_hom}$, there is a non zero formal group homomorphism from $F(X,Y)$ to
    $\bfG_a(X,Y)$ or $\bfG_m(X,Y)$. According to lemma $\ref{lem: target_com}$ and
    $\bfG_a(X,Y), \bfG_m (X,Y)$ is commutative formal group law, then $F(X,Y)$ is commutative. This
    leads to a contradiction.\par
    Thirdly, we prove the sufficiency for the case $R$ is $\ZZ$-torsion free.



\end{proof}
